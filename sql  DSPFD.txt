SELECT FD_ATR.ATFILE, FD_ATR.ATTXT, FD_ACCPTH.APKEYF, FFD.WHFLDB,FFD.WHFTXT 

FROM fd_atr INNER JOIN fd_accpth
USING  (FD_ATR.ATFILE = FD_ACCPTH.APFILE),
fd_accpth INNER JOIN FFD
USING (FD_ACCPTH.APKEYF, FFD.WHFLDE)      
 

WHERE FD_ATR.ATFILE = FD_ACCPTH.APFILE AND FD_ATR.ATFILE = FFD.WHFILE AND                       
 FD_ACCPTH.APKEYF = FFD.WHFLDE      

SELECT EMPNO, ACSTDATE
FROM CORPDATA.PROJACT INNER JOIN CORPDATA.EMPPROJACT
USING (PROJNO, ACTNO)
WHERE ACSDATE > ’1982-12-31’;
